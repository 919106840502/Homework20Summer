//P252 第八题#include <iostream>#include <cstring>using namespace std;class Cat{private: int age; double weight; string color;public: Cat() {  age = 0;  weight = 0.0;  color = "\0"; } void setage(int x) {  age = x; } int getage() {  return age; } void setweight(double x) {  weight = x; } double getweight() {  return weight; } void setcolor(string str) {  color = str; } string getcolor() {  return color; }};int main(){ Cat cat[20]; int age; double weight; string color; int num; int T;  cin>>T; while (T--) {    cin >> num;    cout << "输入年龄：" << endl;    cin >> age;    cat[num - 1].setage(age);    cout << "输入体重：" << endl;    cin >> weight;    cat[num - 1].setweight(weight);    cout << "输入颜色：" << endl;    cin >> color;    cout<<endl;     cat[num - 1].setcolor(color);     age = cat[num - 1].getage();     weight = cat[num - 1].getweight();     color = cat[num - 1].getcolor();     cout << "年龄:" << age << endl;     cout << "体重:" << weight << endl;     cout << "颜色:" << color << endl;     cout << endl;   } return 0;}
